## 一、方案概述

### 1.1 核心架构

```
用户（企微私聊） → 企微消息回调 → 扣子Webhook → Bot处理 → 回复用户
                                              ↓
                                        企微客服消息API
                                              ↓
                                        自动开场白/跟进
```

### 1.2 方案选型

| 方案 | 成本 | 复杂度 | 推荐度 |
|------|------|--------|--------|
| **方案A：企微自建应用+扣子Webhook** | 0元 | 中等 | ⭐⭐⭐⭐⭐ |
| 方案B：企微客服消息API | 0元 | 高 | ⭐⭐⭐ |
| 方案C：第三方SCRM | 付费 | 低 | 不推荐 |

**推荐方案A**：使用企微自建应用接收消息，通过扣子Webhook触发Bot处理，再通过企微客服消息API回复用户。完全0成本，利用企微原生能力。

### 2.3 获取Access Token

#### 方法：通过API获取

使用以下接口获取Access Token：

```
GET https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid=YOUR_CORPID&corpsecret=YOUR_SECRET
```

**Python示例代码：**

```python
import requests
import json

def get_access_token(corp_id, corp_secret):
    """获取企微Access Token"""
    url = f"https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid={corp_id}&corpsecret={corp_secret}"
    response = requests.get(url)
    result = response.json()
    
    if result.get("errcode") == 0:
        return result.get("access_token")
    else:
        raise Exception(f"获取token失败: {result}")

# 使用示例
corp_id = "ww1234567890abcdef"
corp_secret = "your_secret_here"

print(f"Access Token: {token}")
```

```python
import hashlib
import xml.etree.ElementTree as ET
import json
import time
import requests

app = Flask(__name__)

# 配置项
WECHAT_TOKEN = "fly1234567890abcdef"
ENCODING_AES_KEY = "abcdefghijklmnopqrstuvwxyz1234567890ABCD"
CORP_ID = "ww1234567890abcdef"
AGENT_ID = "1000001"
APP_SECRET = "your_secret_here"

# 扣子Bot配置
COZE_BOT_ID = "7634446147243311138"
COZE_API_TOKEN = "sat_TZOQ6PbNVji5YHZoqzr8bFzCWN5I8k7wEDyan9kfaVBfrBladvimJksZynzIZNaq"

# 用户状态存储（生产环境建议用Redis）
user_states = {}  # {openid: {"round": 1, "paid": False, "profile": {...}}}

def get_access_token():
    """获取Access Token"""
    url = f"https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid={CORP_ID}&corpsecret={APP_SECRET}"
    response = requests.get(url)
    result = response.json()
    return result.get("access_token")

def send_message_to_user(openid, content):
    """发送消息给用户"""
    token = get_access_token()
    url = f"https://qyapi.weixin.qq.com/cgi-bin/message/custom/send?access_token={token}"
    
    data = {
        "touser": openid,
        "msgtype": "text",
        "agentid": int(AGENT_ID),
        "text": {
            "content": content
        }
    }
    
    response = requests.post(url, json=data)
    return response.json()

def call_coze_bot(user_id, message):
    """调用扣子Bot"""
    url = "https://api.coze.cn/v3/chat"
    headers = {
        "Authorization": f"Bearer {COZE_API_TOKEN}",
        "Content-Type": "application/json"
    }
    
    data = {
        "bot_id": COZE_BOT_ID,
        "user_id": user_id,
        "query": message,
        "stream": False
    }
    
    response = requests.post(url, headers=headers, json=data)
    return response.json()

def verify_wechat_server(request):
    """验证微信服务器"""
    signature = request.args.get('signature', '')
    timestamp = request.args.get('timestamp', '')
    nonce = request.args.get('nonce', '')
    
    token = WECHAT_TOKEN
    tmp_list = [token, timestamp, nonce]
    tmp_list.sort()
    tmp_str = ''.join(tmp_list)
    hash_obj = hashlib.sha1(tmp_str.encode())
    expected_signature = hash_obj.hexdigest()
    
    return signature == expected_signature

@app.route('/wechat', methods=['GET'])
def wechat_verify():
    """微信服务器验证"""
    if verify_wechat_server(request):
        echostr = request.args.get('echostr', '')
        return Response(echostr, mimetype='text/plain')
    return "验证失败"

@app.route('/wechat', methods=['POST'])
def wechat_message():
    """接收微信消息"""
    # 验证签名
    if not verify_wechat_server(request):
        return "fail"
    
    # 解析消息
    xml_data = request.data
    root = ET.fromstring(xml_data)
    
    msg_type = root.find('MsgType').text
    from_user = root.find('FromUserName').text
    content = root.find('Content').text if root.find('Content') is not None else ""
    
    print(f"[收到消息] 用户: {from_user}, 类型: {msg_type}, 内容: {content}")
    
    # 初始化用户状态
    if from_user not in user_states:
        user_states[from_user] = {
            "round": 0,
            "paid": False,
            "paid_type": None,
            "profile": {}
        }
    
    # 检查是否付费用户
    if not user_states[from_user]["paid"]:
        # 未付费用户，检查轮次
        if user_states[from_user]["round"] >= 3:
            # 已用完3轮免费
            locked_message = """🔒 【体验次数已用完】

• 59元/月

• 999元/年卡

点击下方链接立即解锁 → [购买地址]

或回复「价格」查看详细套餐对比"""
            send_message_to_user(from_user, locked_message)
            return "success"
        
        # 计数+1
        user_states[from_user]["round"] += 1
    
    # 调用扣子Bot处理
    try:
        coze_response = call_coze_bot(from_user, content)
        
        if coze_response.get("code") == 0:
            chat_id = coze_response["data"]["chat_id"]
            
            # 轮询获取回复（简化版，生产环境建议用WebSocket）
            for _ in range(30):  # 最多等30秒
                time.sleep(1)
                # 这里应该调用获取消息接口
                # 为简化，直接使用模拟回复
                pass
            
            # 提取Bot回复
            bot_reply = coze_response.get("data", {}).get("messages", [{}])[0].get("content", "")
            
            # 如果没有回复内容，发送预设消息
            if not bot_reply:
                bot_reply = "感谢您的提问！FLY正在为您准备回复，请稍候..."

        else:
            send_message_to_user(from_user, "抱歉，服务暂时异常，请稍后重试。")

    return "success"

if __name__ == '__main__':
```

#### Step 3：工作流设计

```
[Webhook触发] → [提取用户ID和消息] → [检查用户状态] → [调用Bot处理] → [返回回复]
```

**详细节点配置：**

1. **Webhook触发器**：自动接收企微推送的消息
2. **消息解析**：提取openid、content等字段
3. **状态检查**：查询用户当前轮次和付费状态
4. **Bot处理**：调用扣子Chat API
5. **结果返回**：将Bot回复推送回企微

```python
# 在 wechat_server.py 中添加转发逻辑

def forward_to_coze(message_data):
    """转发消息到扣子"""
    webhook_url = "https://api.coze.cn/v1/workflow/.../trigger"
```

## 五、3轮免费体验计数机制

### 5.1 设计原则

1. **终身1次**：每个用户只有一次免费3轮的机会
2. **本地记录**：使用用户OpenID作为唯一标识
3. **防刷机制**：简单的IP限流

### 5.2 计数逻辑实现

```python
# 用户状态存储结构
user_states = {
    "openid_001": {
        "round": 0,           # 已使用轮次
        "total_free": 3,      # 总免费轮次
        "has_used_free": False,  # 是否使用过免费
        "paid": False,        # 是否付费
        "paid_type": None,    # 付费类型: '7day', 'monthly', 'unlimited', 'yearly'
        "paid_expire": None,  # 付费到期时间
```

```python
def match_case(industry, pain_point, platform, style=None):
    """
    根据条件匹配最佳案例
    
    参数:
        industry: 行业
        pain_point: 痛点
        platform: 平台
        style: 风格偏好
    
    返回:
        matched_cases: 匹配的案例列表
    """
    
    results = []
    
    # 精确匹配行业
    if industry in CASE_LIBRARY:
        for sub_category, cases in CASE_LIBRARY[industry].items():
            for case in cases:
                score = 0
                
                # 平台匹配
                if platform and case["platform"] == platform:
                    score += 3
                
                # 风格匹配
                if style and case["style"] == style:
                    score += 2
                
                # 痛点相关（关键词匹配）
                if pain_point:
                    case_text = case["copy"] + case["name"]
                    if any(p in case_text for p in pain_point if isinstance(pain_point, list)):
                        score += 1
                    elif pain_point in case_text:
                        score += 1
                
                if score > 0:
                    results.append((case, score))
    
    # 按分数排序
    results.sort(key=lambda x: x[1], reverse=True)
    
    # 返回前3个最佳案例
    return [case for case, score in results[:3]]

def generate_case_intro(matched_cases):
    """生成案例推荐话术"""
    
    if not matched_cases:
        return ""
    
    intro = "\n\n📊 参考同行案例：\n" + "=" * 40 + "\n\n"
    
    for i, case in enumerate(matched_cases, 1):
        intro += f"{i}. 【{case['name']}】\n"
        intro += f"   📍 平台：{case['platform']}\n"
        intro += f"   📝 文案：{case['copy']}\n"
        intro += f"   ✨ 效果：{case['effect']}\n\n"
    
    intro += "=" * 40
    intro += "\n\n💡 您想要类似的效果吗？我可以帮您生成专属文案~"
    
    return intro
```

```python
def extract_profile_from_message(message, current_profile):
    """
    从消息中提取客户信息并更新档案
    
    参数:

        current_profile: 当前档案
    
    返回:
        updated_profile: 更新后的档案
    """
    
    profile = current_profile.copy()
    
    # 行业关键词
    industry_patterns = {
        "火锅": "餐饮", "烧烤": "餐饮", "奶茶": "餐饮", "咖啡": "餐饮",
        "烘焙": "餐饮", "蛋糕": "餐饮", "快餐": "餐饮", "小龙虾": "餐饮",

        "服装": "零售", "鞋": "零售", "箱包": "零售",

        "少儿": "教育", "英语": "教育", "培训": "教育", "驾校": "教育",
        "宠物": "服务业", "宠物店": "服务业",
    }
    
    # 提取行业
    for keyword, industry in industry_patterns.items():
        if keyword in message:
            profile["industry"] = industry
            break
    
    # 提取店铺名称
    shop_patterns = [
        r'(?:店名|叫|名字)[是为]?([^\s，,。!！]+)',
        r'我是([^\s，,。!！]+)',
        r'在([^\s，,。!！]+)开了?个?店',
    ]
    for pattern in shop_patterns:
        match = re.search(pattern, message)
        if match:
            potential_shop = match.group(1)
            if len(potential_shop) >= 2 and len(potential_shop) <= 20:
                profile["shop_name"] = potential_shop
                break
    
    # 提取地址
    address_patterns = [
        r'(?:地址|位置)在([^\s，,。!！]+)',
        r'在(.+?)[的市县区镇]',
    ]
    for pattern in address_patterns:
        match = re.search(pattern, message)
        if match:
            potential_address = match.group(1)
            if len(potential_address) >= 3:
                profile["shop_address"] = potential_address
                break
    
    # 提取痛点
    pain_point_keywords = {
        "曝光": ["曝光", "没人知道", "没流量", "没客流"],
        "转化": ["转化", "留不住", "成交", "卖不动"],

        "活动": ["活动", "促销", "优惠", "节日"],
        "内容": ["内容", "文案", "发圈", "没素材"],
        "引流": ["引流", "拓客", "拉新", "获客"],
    }
    
    detected_pains = []
    for pain, keywords in pain_point_keywords.items():
        if any(kw in message for kw in keywords):
            detected_pains.append(pain)
    
    if detected_pains:
        existing_pains = profile.get("pain_points", [])
        profile["pain_points"] = list(set(existing_pains + detected_pains))
    
    # 提取风格偏好
    style_keywords = {
        "温情": ["温情", "温暖", "感人", "妈妈", "亲情"],
        "活力": ["活力", "青春", "年轻", "激情"],
        "专业": ["专业", "严谨", "权威"],
        "搞笑": ["搞笑", "幽默", "段子"],
        "精致": ["精致", "高端", "奢华"],
        "接地气": ["接地气", "亲民", "实在"],
    }
    
    for style, keywords in style_keywords.items():
        if any(kw in message for kw in keywords):
            profile["preferred_style"] = style
            break
    
    # 提取平台偏好
    platform_keywords = {
        "朋友圈": ["朋友圈", "微信"],
        "小红书": ["小红书", "种草"],
        "抖音": ["抖音", "短视频"],
        "美团": ["美团", "团购"],
    }
    
    for platform, keywords in platform_keywords.items():
        if any(kw in message for kw in keywords):
            profile["preferred_platform"] = platform
            break
    
    # 更新时间
    profile["last_updated"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    
    return profile
```
